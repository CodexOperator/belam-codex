8pub enum Operand<T> {
9    Column(ColumnRef<T>),
10    Literal(LiteralValue),

13pub enum BoolExpr<T> {
14    Eq(Operand<T>, Operand<T>),
15    Ne(Operand<T>, Operand<T>),
16    Gt(Operand<T>, Operand<T>),
17    Lt(Operand<T>, Operand<T>),
18    Gte(Operand<T>, Operand<T>),
19    Lte(Operand<T>, Operand<T>),
20    And(Box<BoolExpr<T>>, Box<BoolExpr<T>>),
21    Or(Box<BoolExpr<T>>, Box<BoolExpr<T>>),
22}
23
24impl<T> BoolExpr<T> {
25    pub fn and(self, other: BoolExpr<T>) -> BoolExpr<T> {
26        BoolExpr::And(Box::new(self), Box::new(other))
27    }
28
29    pub fn or(self, other: BoolExpr<T>) -> BoolExpr<T> {
30        BoolExpr::Or(Box::new(self), Box::new(other))
31    }
32}

34/// Trait for types that can be used as the right-hand side of a comparison with a column of type V
35/// in table T.
36///
37/// This trait is implemented for Col<T, V> and various literal types.
38pub trait RHS<T, V> {
39    fn to_expr(self) -> Operand<T>;
40}
41
42impl<T, V> RHS<T, V> for Col<T, V> {
43    fn to_expr(self) -> Operand<T> {
44        Operand::Column(self.col)
45    }
46}
47
48fn format_bool_expr<T>(v: &Operand<T>) -> String {
49    match v {
50        Operand::Column(col) => col.fmt(),
51        Operand::Literal(lit) => lit.0.clone(),
52    }
53}
54
55pub fn format_expr<T>(expr: &BoolExpr<T>) -> String {
56    match expr {
57        BoolExpr::Eq(l, r) => format!("({} = {})", format_bool_expr(l), format_bool_expr(r)),
58        BoolExpr::Ne(l, r) => format!("({} <> {})", format_bool_expr(l), format_bool_expr(r)),
59        BoolExpr::Gt(l, r) => format!("({} > {})", format_bool_expr(l), format_bool_expr(r)),
60        BoolExpr::Lt(l, r) => format!("({} < {})", format_bool_expr(l), format_bool_expr(r)),
61        BoolExpr::Gte(l, r) => format!("({} >= {})", format_bool_expr(l), format_bool_expr(r)),
62        BoolExpr::Lte(l, r) => format!("({} <= {})", format_bool_expr(l), format_bool_expr(r)),
63        BoolExpr::And(a, b) => format!("({} AND {})", format_expr(a), format_expr(b)),
64        BoolExpr::Or(a, b) => format!("({} OR {})", format_expr(a), format_expr(b)),
65    }
66}

77macro_rules! impl_rhs {
78    ($ty:ty, $formatter:expr) => {
79        impl<T> RHS<T, $ty> for $ty {
80            fn to_expr(self) -> Operand<T> {
81                Operand::Literal(LiteralValue($formatter(self)))
82            }
83        }
84    };
85}
