31        let (TableAccess::Public(span) | TableAccess::Private(span)) = *self;

68impl TableArgs {
69    pub(crate) fn parse(input: TokenStream, struct_ident: &Ident) -> syn::Result<Self> {
70        let mut access = None;
71        let mut scheduled = None;
72        let mut name = None;
73        let mut indices = Vec::new();
74        syn::meta::parser(|meta| {
75            match_meta!(match meta {
76                sym::public => {
77                    check_duplicate_msg(&access, &meta, "already specified access level")?;
78                    access = Some(TableAccess::Public(meta.path.span()));
79                }
80                sym::private => {
81                    check_duplicate_msg(&access, &meta, "already specified access level")?;
82                    access = Some(TableAccess::Private(meta.path.span()));
83                }
84                sym::name => {
85                    check_duplicate(&name, &meta)?;
86                    let value = meta.value()?;
87                    name = Some(value.parse()?);
88                }
89                sym::index => indices.push(IndexArg::parse_meta(meta)?),
90                sym::scheduled => {
91                    check_duplicate(&scheduled, &meta)?;
92                    scheduled = Some(ScheduledArg::parse_meta(meta)?);
93                }
94            });
95            Ok(())
96        })
97        .parse2(input)?;
98        let name = name.ok_or_else(|| {
99            let table = struct_ident.to_string().to_snake_case();
100            syn::Error::new(
101                Span::call_site(),
102                format_args!("must specify table name, e.g. `#[spacetimedb::table(name = {table})]"),
103            )
104        })?;
105        Ok(TableArgs {
106            access,
107            scheduled,
108            name,
109            indices,
110        })
111    }
112}

114impl ScheduledArg {
115    fn parse_meta(meta: ParseNestedMeta) -> syn::Result<Self> {
116        let span = meta.path.span();
117        let mut reducer_or_procedure = None;
118        let mut at = None;
119
120        meta.parse_nested_meta(|meta| {
121            if meta.input.peek(syn::Token![=]) || meta.input.peek(syn::token::Paren) {
122                match_meta!(match meta {
123                    sym::at => {
124                        check_duplicate(&at, &meta)?;
125                        let ident = meta.value()?.parse()?;
126                        at = Some(ident);
127                    }
128                })
129            } else {
130                check_duplicate_msg(
131                    &reducer_or_procedure,
132                    &meta,
133                    "can only specify one scheduled reducer or procedure",
134                )?;
135                reducer_or_procedure = Some(meta.path);
136            }
137            Ok(())
138        })?;
139
140        let reducer_or_procedure = reducer_or_procedure.ok_or_else(|| {
141            meta.error(
142                "must specify scheduled reducer or procedure associated with the table: scheduled(function_name)",
143            )
144        })?;
145        Ok(Self {
146            span,
147            reducer_or_procedure,
148            at,
149        })
150    }

153impl IndexArg {
154    fn parse_meta(meta: ParseNestedMeta) -> syn::Result<Self> {
155        let mut name = None;
156        let mut algo = None;
157
158        meta.parse_nested_meta(|meta| {
159            match_meta!(match meta {
160                sym::name => {
161                    check_duplicate(&name, &meta)?;
162                    name = Some(meta.value()?.parse()?);
163                }
164                sym::btree => {
165                    check_duplicate_msg(&algo, &meta, "index algorithm specified twice")?;
166                    algo = Some(Self::parse_btree(meta)?);
167                }
168                sym::hash => {
169                    check_duplicate_msg(&algo, &meta, "index algorithm specified twice")?;
170                    algo = Some(Self::parse_hash(meta)?);
171                }
172                sym::direct => {
173                    check_duplicate_msg(&algo, &meta, "index algorithm specified twice")?;
174                    algo = Some(Self::parse_direct(meta)?);
175                }
176            });
177            Ok(())
178        })?;
179        let name = name.ok_or_else(|| meta.error("missing index name, e.g. name = my_index"))?;
180        let kind = algo.ok_or_else(|| {
181            meta.error(
182                "missing index algorithm, e.g., `btree(columns = [col1, col2])`, \
183                `hash(columns = [col1, col2])`, or `direct(column = col1)`",
184            )
185        })?;
186
187        Ok(IndexArg::new(name, kind))
188    }
189
190    fn parse_columns(meta: &ParseNestedMeta) -> syn::Result<Option<Vec<Ident>>> {
191        let mut columns = None;
192        meta.parse_nested_meta(|meta| {
193            match_meta!(match meta {
194                sym::columns => {
195                    check_duplicate(&columns, &meta)?;
196                    let value = meta.value()?;
197                    let inner;
198                    syn::bracketed!(inner in value);
199                    columns = Some(
200                        Punctuated::<Ident, Token![,]>::parse_terminated(&inner)?
201                            .into_iter()
202                            .collect::<Vec<_>>(),
203                    );
204                }
205            });
206            Ok(())
207        })?;
208        Ok(columns)
209    }
210
211    fn parse_btree(meta: ParseNestedMeta) -> syn::Result<IndexType> {
212        let columns = Self::parse_columns(&meta)?;
213        let columns = columns
214            .ok_or_else(|| meta.error("must specify columns for btree index, e.g. `btree(columns = [col1, col2])`"))?;
215        Ok(IndexType::BTree { columns })
216    }
217
218    fn parse_hash(meta: ParseNestedMeta) -> syn::Result<IndexType> {
219        let columns = Self::parse_columns(&meta)?;
220        let columns = columns
221            .ok_or_else(|| meta.error("must specify columns for hash index, e.g. `hash(columns = [col1, col2])`"))?;
222        Ok(IndexType::Hash { columns })
223    }
224
225    fn parse_direct(meta: ParseNestedMeta) -> syn::Result<IndexType> {
226        let mut column = None;
227        meta.parse_nested_meta(|meta| {
228            match_meta!(match meta {
229                sym::column => {
230                    check_duplicate(&column, &meta)?;
231                    let value = meta.value()?;
232                    let inner;
233                    syn::bracketed!(inner in value);
234                    column = Some(Ident::parse(&inner)?);
235                }
236            });
237            Ok(())
238        })?;
239        let column = column
240            .ok_or_else(|| meta.error("must specify the column for direct index, e.g. `direct(column = col1)`"))?;
241        Ok(IndexType::Direct { column })
242    }

244    /// Parses an inline `#[index(btree)]`, `#[index(hash)]`, or `#[index(direct)]` attribute on a field.
245    fn parse_index_attr(field: &Ident, attr: &syn::Attribute) -> syn::Result<Self> {
246        let mut kind = None;
247        attr.parse_nested_meta(|meta| {
248            match_meta!(match meta {
249                sym::btree => {
250                    check_duplicate_msg(&kind, &meta, "index type specified twice")?;
251                    kind = Some(IndexType::BTree {
252                        columns: vec![field.clone()],
253                    });
254                }
255                sym::hash => {
256                    check_duplicate_msg(&kind, &meta, "index type specified twice")?;
257                    kind = Some(IndexType::Hash {
258                        columns: vec![field.clone()],
259                    });
260                }
261                sym::direct => {
262                    check_duplicate_msg(&kind, &meta, "index type specified twice")?;
263                    kind = Some(IndexType::Direct { column: field.clone() })
264                }
265            });
266            Ok(())
267        })?;
268        let kind = kind
269            .ok_or_else(|| syn::Error::new_spanned(&attr.meta, "must specify kind of index (`btree` or `direct`)"))?;
270        let name = field.clone();
271        Ok(IndexArg::new(name, kind))
272    }

274    fn validate<'a>(&'a self, table_name: &str, cols: &'a [Column<'a>]) -> syn::Result<ValidatedIndex<'a>> {
275        let find_column = |ident| find_column(cols, ident);
276        let (kind, kind_str) = match &self.kind {
277            IndexType::BTree { columns } => {
278                let cols = columns.iter().map(find_column).collect::<syn::Result<Vec<_>>>()?;
279                (ValidatedIndexType::BTree { cols }, "btree")
280            }
281            IndexType::Hash { columns } => {
282                let cols = columns.iter().map(find_column).collect::<syn::Result<Vec<_>>>()?;
283                (ValidatedIndexType::Hash { cols }, "hash")
284            }
285            IndexType::Direct { column } => {
286                let col = find_column(column)?;
287
288                if !self.is_unique {
289                    return Err(syn::Error::new(
290                        column.span(),
291                        "a direct index must be paired with a `#[unique] constraint",
292                    ));
293                }
294
295                (ValidatedIndexType::Direct { col }, "direct")
296            }
297        };
298        // See crates/schema/src/validate/v9.rs for the format of index names.
299        // It's slightly unnerving that we just trust that component to generate this format correctly,
300        // but what can you do.
301        let cols = kind.columns();
302        let cols = cols.iter().map(|col| col.ident.to_string()).collect::<Vec<_>>();
303        let cols = cols.join("_");
304        let index_name = format!("{table_name}_{cols}_idx_{kind_str}");
305
306        Ok(ValidatedIndex {
307            is_unique: self.is_unique,
308            index_name,
309            accessor_name: &self.name,
310            kind,
311        })
312    }

371enum ValidatedIndexType<'a> {
372    BTree { cols: Vec<&'a Column<'a>> },
373    Hash { cols: Vec<&'a Column<'a>> },
374    Direct { col: &'a Column<'a> },

377impl ValidatedIndexType<'_> {
378    fn columns(&self) -> &[&Column<'_>] {
379        match self {
380            Self::BTree { cols } | Self::Hash { cols } => cols,
381            Self::Direct { col } => slice::from_ref(col),
382        }
383    }
384
385    fn one_col(&self) -> Option<&Column<'_>> {
386        match self.columns() {
387            [col] => Some(col),
388            _ => None,
389        }
390    }

393impl ValidatedIndex<'_> {
394    fn desc(&self) -> TokenStream {
395        let algo = match &self.kind {
396            ValidatedIndexType::BTree { cols } => {
397                let col_ids = cols.iter().map(|col| col.index);
398                quote!(spacetimedb::table::IndexAlgo::BTree {
399                    columns: &[#(#col_ids),*]
400                })
401            }
402            ValidatedIndexType::Hash { cols } => {
403                let col_ids = cols.iter().map(|col| col.index);
404                quote!(spacetimedb::table::IndexAlgo::Hash {
405                    columns: &[#(#col_ids),*]
406                })
407            }
408            ValidatedIndexType::Direct { col } => {
409                let col_id = col.index;
410                quote!(spacetimedb::table::IndexAlgo::Direct {
411                    column: #col_id
412                })
413            }
414        };
415        let accessor_name = ident_to_litstr(self.accessor_name);
416        // Note: we do not pass the index_name through here.
417        // We trust the schema validation logic to reconstruct the name we've stored in `self.name`.
418        quote!(spacetimedb::table::IndexDesc {
419            accessor_name: #accessor_name,
420            algo: #algo,
421        })
422    }
423
424    fn accessor(
425        &self,
426        vis: &syn::Visibility,
427        row_type_ident: &Ident,
428        tbl_type_ident: &Ident,
429        flavor: AccessorType,
430    ) -> TokenStream {
431        if self.is_unique {
432            self.unique_accessor(row_type_ident, tbl_type_ident, flavor)
433        } else {
434            self.non_unique_accessor(vis, row_type_ident, tbl_type_ident, flavor)
435        }
436    }
437
438    fn unique_accessor(&self, row_type_ident: &Ident, tbl_type_ident: &Ident, flavor: AccessorType) -> TokenStream {
439        let col = self.kind.one_col().unwrap();
440        let index_ident = self.accessor_name;
441        let vis = col.vis;
442        let col_ty = col.ty;
443        let column_ident = col.ident;
444
445        let unique_ty = flavor.unique();
446        let tbl_token = quote!(#tbl_type_ident);
447        let doc_type = flavor.unique_doc_typename();
448
449        let doc = format!(
450            "Gets the [`{doc_type}`][spacetimedb::{doc_type}] for the \
451             [`{column_ident}`][{row_type_ident}::{column_ident}] column."
452        );
453        quote! {
454            #[doc = #doc]
455            #vis fn #column_ident(&self) -> #unique_ty<#tbl_token, #col_ty, __indices::#index_ident> {
456                #unique_ty::__NEW
457            }
458        }
459    }
460
461    fn non_unique_accessor(
462        &self,
463        vis: &syn::Visibility,
464        row_type_ident: &Ident,
465        tbl_type_ident: &Ident,
466        flavor: AccessorType,
467    ) -> TokenStream {
468        let index_ident = self.accessor_name;
469        let cols = self.kind.columns();
470        let col_tys = cols.iter().map(|c| c.ty);
471
472        let (handle_ty, doc_type, kind_doc) = match &self.kind {
473            ValidatedIndexType::BTree { .. } => (flavor.range(), flavor.range_doc_typename(), "B-tree"),
474            // Should be unreachable, but we might as well include this.
475            ValidatedIndexType::Direct { .. } => (flavor.range(), flavor.range_doc_typename(), "Direct"),
476            ValidatedIndexType::Hash { .. } => (flavor.point(), flavor.point_doc_typename(), "Hash"),
477        };
478        let mut doc = format!(
479            "Gets the `{index_ident}` [`{doc_type}`][spacetimedb::{doc_type}] as defined \
480                    on this table.\n\nThis {kind_doc} index is defined on the following columns, in order:\n"
481        );
482        for col in cols {
483            use std::fmt::Write;
484            writeln!(
485                doc,
486                "- [`{ident}`][{row_type_ident}#structfield.{ident}]: [`{ty}`]",
487                ident = col.ident,
488                ty = col.ty.to_token_stream()
489            )
490            .unwrap();
491        }
492
493        let tbl_token = quote!(#tbl_type_ident);
494        quote! {
495            #[doc = #doc]
496            #vis fn #index_ident(&self) -> #handle_ty<#tbl_token, (#(#col_tys,)*), __indices::#index_ident> {
497                #handle_ty::__NEW
498            }
499        }
500    }

502    fn marker_type(&self, vis: &syn::Visibility, tablehandle_ident: &Ident) -> TokenStream {
503        let index_ident = self.accessor_name;
504        let index_name = &self.index_name;
505
506        let (typeck_direct_index, is_ranged) = match &self.kind {
507            ValidatedIndexType::BTree { .. } => (None, true),
508            ValidatedIndexType::Direct { col } => {
509                let col_ty = col.ty;
510                let typeck = quote_spanned!(col_ty.span()=>
511                    const _: () = {
512                        spacetimedb::spacetimedb_lib::assert_column_type_valid_for_direct_index::<#col_ty>();
513                    };
514                );
515                (Some(typeck), true)
516            }
517            ValidatedIndexType::Hash { .. } => (None, false),
518        };
519        let vis = if self.is_unique {
520            self.kind.one_col().unwrap().vis
521        } else {
522            vis
523        };
524        let vis = superize_vis(vis);
525
526        let cols = self.kind.columns();
527        let num_cols = cols.len();
528        let index_kind_trait = if is_ranged {
529            quote!(IndexIsRanged)
530        } else {
531            quote!(IndexIsPointed)
532        };
533        let mut decl = quote! {
534            #typeck_direct_index
535
536            #vis struct #index_ident;
537            impl spacetimedb::table::#index_kind_trait for #index_ident {}
538            impl spacetimedb::table::Index for #index_ident {
539                const NUM_COLS_INDEXED: usize = #num_cols;
540                fn index_id() -> spacetimedb::table::IndexId {
541                    static INDEX_ID: std::sync::OnceLock<spacetimedb::table::IndexId> = std::sync::OnceLock::new();
542                    *INDEX_ID.get_or_init(|| {
543                        spacetimedb::sys::index_id_from_name(#index_name).unwrap()
544                    })
545                }
546            }
547        };
548        if self.is_unique {
549            let col = self.kind.one_col().unwrap();
550            let col_ty = col.ty;
551            let col_name = col.ident.to_string();
552            let field_ident = col.ident;
553            decl.extend(quote! {
554                impl spacetimedb::table::Column for #index_ident {
555                    type Table = #tablehandle_ident;
556                    type ColType = #col_ty;
557                    const COLUMN_NAME: &'static str = #col_name;
558                    fn get_field(row: &<Self::Table as spacetimedb::Table>::Row) -> &Self::ColType {
559                        &row.#field_ident
560                    }
561                }
562            });
563        }
564        decl
565    }
566}

568/// Transform a visibility marker to one with the same effective visibility, but
569/// for use in a child module of the module of the original marker.
570fn superize_vis(vis: &syn::Visibility) -> Cow<'_, syn::Visibility> {
571    match vis {
572        syn::Visibility::Public(_) => Cow::Borrowed(vis),
573        syn::Visibility::Restricted(vis_r) => {
574            let first = &vis_r.path.segments[0];
575            if first.ident == "crate" || vis_r.path.leading_colon.is_some() {
576                Cow::Borrowed(vis)
577            } else {
578                let mut vis_r = vis_r.clone();
579                if first.ident == "super" {
580                    vis_r.path.segments.insert(0, first.clone())
581                } else if first.ident == "self" {
582                    vis_r.path.segments[0].ident = Ident::new("super", Span::call_site())
583                }
584                Cow::Owned(syn::Visibility::Restricted(vis_r))
585            }
586        }
587        syn::Visibility::Inherited => Cow::Owned(parse_quote!(pub(super))),
588    }
589}

600fn try_find_column<'a, 'b, T: ?Sized>(cols: &'a [Column<'b>], name: &T) -> Option<&'a Column<'b>>
601where
602    Ident: PartialEq<T>,
603{
604    cols.iter().find(|col| col.ident == name)
605}

607fn find_column<'a, 'b>(cols: &'a [Column<'b>], name: &Ident) -> syn::Result<&'a Column<'b>> {
608    try_find_column(cols, name).ok_or_else(|| syn::Error::new(name.span(), "not a column of the table"))
609}

611enum ColumnAttr {
612    Unique(Span),
613    AutoInc(Span),
614    PrimaryKey(Span),
615    Index(IndexArg),
616    Default(syn::Expr, Span),
617}
618
619impl ColumnAttr {
620    fn parse(attr: &syn::Attribute, field_ident: &Ident) -> syn::Result<Option<Self>> {
621        let Some(ident) = attr.path().get_ident() else {
622            return Ok(None);
623        };
624        Ok(if ident == sym::index {
625            let index = IndexArg::parse_index_attr(field_ident, attr)?;
626            Some(ColumnAttr::Index(index))
627        } else if ident == sym::unique {
628            attr.meta.require_path_only()?;
629            Some(ColumnAttr::Unique(ident.span()))
630        } else if ident == sym::auto_inc {
631            attr.meta.require_path_only()?;
632            Some(ColumnAttr::AutoInc(ident.span()))
633        } else if ident == sym::primary_key {
634            attr.meta.require_path_only()?;
635            Some(ColumnAttr::PrimaryKey(ident.span()))
636        } else if ident == sym::default {
637            Some(parse_default_attr(attr, ident)?)
638        } else {
639            None
640        })
641    }
642}
643
644fn parse_default_attr(attr: &syn::Attribute, ident: &Ident) -> syn::Result<ColumnAttr> {
645    if let Ok(expr) = attr.parse_args::<syn::Expr>() {
646        return Ok(ColumnAttr::Default(expr, ident.span()));
647    }
648
649    Err(syn::Error::new_spanned(
650        &attr.meta,
651        "expected default value in format `#[default(CONSTANT_VALUE)]`",
652    ))
653}

662    std::sync::LazyLock::new(|| Mutex::new(HashSet::new()));

664fn is_first_appearance(struct_name: &str) -> bool {
665    let mut set = GENERATED_STRUCTS.lock().expect("mutex poisoned");
666
667    set.insert(struct_name.to_string())
668}
669
670pub(crate) fn table_impl(mut args: TableArgs, item: &syn::DeriveInput) -> syn::Result<TokenStream> {
671    let vis = &item.vis;
672    let sats_ty = sats::sats_type_from_derive(item, quote!(spacetimedb::spacetimedb_lib))?;
673
674    let original_struct_ident = sats_ty.ident;
675    let table_ident = &args.name;
676    let view_trait_ident = format_ident!("{}__view", table_ident);
677    let query_trait_ident = format_ident!("{}__query", table_ident);
678    let query_cols_struct = format_ident!("{}Cols", original_struct_ident);
679    let query_ix_cols_struct = format_ident!("{}IxCols", original_struct_ident);
680    let table_name = table_ident.unraw().to_string();
681    let sats::SatsTypeData::Product(fields) = &sats_ty.data else {
682        return Err(syn::Error::new(Span::call_site(), "spacetimedb table must be a struct"));
683    };
684
685    for param in &item.generics.params {
686        let err = |msg| syn::Error::new_spanned(param, msg);
687        match param {
688            syn::GenericParam::Lifetime(_) => {}
689            syn::GenericParam::Type(_) => return Err(err("type parameters are not allowed on tables")),
690            syn::GenericParam::Const(_) => return Err(err("const parameters are not allowed on tables")),
691        }
692    }
693
694    let table_id_from_name_func = quote! {
695        fn table_id() -> spacetimedb::TableId {
696            static TABLE_ID: std::sync::OnceLock<spacetimedb::TableId> = std::sync::OnceLock::new();
697            *TABLE_ID.get_or_init(|| {
698                spacetimedb::table_id_from_name(<Self as spacetimedb::table::TableInternal>::TABLE_NAME)
699            })
700        }
701    };
702
703    if fields.len() > u16::MAX.into() {
704        return Err(syn::Error::new_spanned(
705            item,
706            "too many columns; the most a table can have is 2^16",
707        ));
708    }
709
710    let mut columns = vec![];
711    let mut unique_columns = vec![];
712    let mut sequenced_columns = vec![];
713    let mut primary_key_column = None;
714
715    for (i, field) in fields.iter().enumerate() {
716        let col_num = i as u16;
717        let field_ident = field.ident.unwrap();
718
719        let mut unique = None;
720        let mut auto_inc = None;
721        let mut primary_key = None;
722        let mut default_value = None;
723        for attr in field.original_attrs {
724            let Some(attr) = ColumnAttr::parse(attr, field_ident)? else {
725                continue;
726            };
727            match attr {
728                ColumnAttr::Unique(span) => {
729                    check_duplicate(&unique, span)?;
730                    unique = Some(span);
731                }
732                ColumnAttr::AutoInc(span) => {
733                    check_duplicate(&auto_inc, span)?;
734                    auto_inc = Some(span);
735                }
736                ColumnAttr::PrimaryKey(span) => {
737                    check_duplicate(&primary_key, span)?;
738                    primary_key = Some(span);
739                }
740                ColumnAttr::Index(index_arg) => args.indices.push(index_arg),
741                ColumnAttr::Default(expr, span) => {
742                    check_duplicate(&default_value, span)?;
743                    default_value = Some(expr);
744                }
745            }
746        }
747
748        if let Some(default_value) = &default_value {
749            if auto_inc.is_some() || primary_key.is_some() || unique.is_some() {
750                return Err(syn::Error::new(
751                    default_value.span(),
752                    "invalid combination: auto_inc, unique index or primary key cannot have a default value",
753                ));
754            };
755        }
756
757        let column = Column {
758            index: col_num,
759            ident: field_ident,
760            vis: field.vis,
761            ty: field.ty,
762            default_value,
763        };
764
765        if unique.is_some() || primary_key.is_some() {
766            unique_columns.push(column.clone());
767        }
768        if auto_inc.is_some() {
769            sequenced_columns.push(column.clone());
770        }
771        if let Some(span) = primary_key {
772            check_duplicate_msg(&primary_key_column, span, "can only have one primary key per table")?;
773            primary_key_column = Some(column.clone());
774        }
775
776        columns.push(column.clone());
777    }
778
779    let row_type = quote!(#original_struct_ident);
780
781    // Mark all indices with a single column matching a unique constraint as unique.
782    // For all the unpaired unique columns, create a unique index.
783    for unique_col in &unique_columns {
784        if args.indices.iter_mut().any(|index| {
785            let covered_by_index = match &index.kind {
786                IndexType::BTree { columns } | IndexType::Hash { columns } => {
787                    &**columns == slice::from_ref(unique_col.ident)
788                }
789                IndexType::Direct { column } => column == unique_col.ident,
790            };
791            index.is_unique |= covered_by_index;
792            covered_by_index
793        }) {
794            continue;
795        }
796        // NOTE(centril): We pick `btree` here if the user does not specify otherwise,
797        // as it's the safest choice of index for the general case,
798        // even if isn't optimal in specific cases.
799        let name = unique_col.ident.clone();
800        let columns = vec![name.clone()];
801        args.indices.push(IndexArg {
802            name,
803            is_unique: true,
804            kind: IndexType::BTree { columns },
805        })
806    }
807
808    let mut indices = args
809        .indices
810        .iter()
811        .map(|index| index.validate(&table_name, &columns))
812        .collect::<syn::Result<Vec<_>>>()?;
813
814    // Order unique accessors before index accessors.
815    indices.sort_by_key(|index| !index.is_unique);
816
817    let tablehandle_ident = format_ident!("{}__TableHandle", table_ident);
818    let viewhandle_ident = format_ident!("{}__ViewHandle", table_ident);
819
820    let index_descs = indices.iter().map(|index| index.desc());
821    let index_accessors_rw = indices
822        .iter()
823        .map(|index| index.accessor(vis, original_struct_ident, &tablehandle_ident, AccessorType::ReadWrite));
824    let index_accessors_ro = indices
825        .iter()
826        .map(|index| index.accessor(vis, original_struct_ident, &tablehandle_ident, AccessorType::Read));
827    let index_marker_types = indices.iter().map(|index| index.marker_type(vis, &tablehandle_ident));
828
829    // Generate `integrate_generated_columns`
830    // which will integrate all generated auto-inc col values into `_row`.
831    let integrate_gen_col = sequenced_columns.iter().map(|col| {
832        let field = col.ident;
833        quote_spanned!(field.span()=>
834            spacetimedb::table::SequenceTrigger::maybe_decode_into(&mut __row.#field, &mut __generated_cols);
835        )
836    });
837    let integrate_generated_columns = quote_spanned!(item.span() =>
838        fn integrate_generated_columns(__row: &mut #row_type, mut __generated_cols: &[u8]) {
839            #(#integrate_gen_col)*
840        }
841    );
842
843    let table_access = args.access.iter().map(|acc| acc.to_value());
844    let unique_col_ids = unique_columns.iter().map(|col| col.index);
845    let primary_col_id = primary_key_column.clone().into_iter().map(|col| col.index);
846    let sequence_col_ids = sequenced_columns.iter().map(|col| col.index);
847
848    let default_type_check: TokenStream = columns
849        .iter()
850        .filter_map(|col| {
851            if let Some(val) = &col.default_value {
852                let ty = &col.ty;
853                let ident_span = col.ident.span();
854                Some(quote_spanned! { ident_span =>
855                    // This closure enforces that `val` is of type `ty` at compile-time.
856                    let _check: #ty = #val;
857                })
858            } else {
859                None
860            }
861        })
862        .collect();
863
864    let col_defaults: Vec<TokenStream> = columns.iter().filter_map(|col| {
865        if let Some(val) = &col.default_value {
866            let col_id = col.index;
867            Some(quote! {
868                spacetimedb::table::ColumnDefault {
869                    col_id: #col_id,
870                    value: #val.serialize(spacetimedb::sats::algebraic_value::ser::ValueSerializer).expect("default value serialization failed"),
871                },
872            })
873        } else {
874            None
875        }
876    }).collect();
877
878    let default_fn: TokenStream = quote! {
879        fn get_default_col_values() -> Vec<spacetimedb::table::ColumnDefault> {
880            [#(#col_defaults)*].to_vec()
881        }
882    };
883
884    let (schedule, schedule_typecheck) = args
885        .scheduled
886        .as_ref()
887        .map(|sched| {
888            let scheduled_at_column = match &sched.at {
889                Some(at) => Some(find_column(&columns, at)?),
890                None => try_find_column(&columns, "scheduled_at"),
891            };
892            // better error message when both are missing
893            if scheduled_at_column.is_none() && primary_key_column.is_none() {
894                return Err(syn::Error::new(
895                    sched.span,
896                    "scheduled table missing required columns; add these to your struct:\n\
897                             #[primary_key]\n\
898                             #[auto_inc]\n\
899                             scheduled_id: u64,\n\
900                             scheduled_at: spacetimedb::ScheduleAt,",
901                ));
902            }
903            let scheduled_at_column = scheduled_at_column.ok_or_else(|| {
904                syn::Error::new(
905                    sched.span,
906                    "scheduled tables must have a `scheduled_at: spacetimedb::ScheduleAt` column. \
907                             if the column has a name besides `scheduled_at`, you can specify it with \
908                             `scheduled(my_reducer, at = custom_scheduled_at)`",
909                )
910            })?;
911            let primary_key_column = primary_key_column.ok_or_else(|| {
912                syn::Error::new(
913                    sched.span,
914                    "scheduled tables must have a `#[primary_key] #[auto_inc] scheduled_id: u64` column",
915                )
916            })?;
917
918            let reducer_or_procedure = &sched.reducer_or_procedure;
919            let scheduled_at_id = scheduled_at_column.index;
920            let desc = quote!(spacetimedb::table::ScheduleDesc {
921                reducer_or_procedure_name: <#reducer_or_procedure as spacetimedb::rt::FnInfo>::NAME,
922                scheduled_at_column: #scheduled_at_id,
923            });
924
925            let primary_key_ty = primary_key_column.ty;
926            let scheduled_at_ty = scheduled_at_column.ty;
927            let typecheck = quote! {
928                spacetimedb::rt::scheduled_typecheck::<
929                    #original_struct_ident,
930                    <#reducer_or_procedure as spacetimedb::rt::FnInfo>::FnKind,
931                >(#reducer_or_procedure);
932                spacetimedb::rt::assert_scheduled_table_primary_key::<#primary_key_ty>();
933                let _ = |x: #scheduled_at_ty| { let _: spacetimedb::ScheduleAt = x; };
934            };
935
936            Ok((desc, typecheck))
937        })
938        .transpose()?
939        .unzip();
940    let schedule = schedule.into_iter();
941
942    let unique_err = if !unique_columns.is_empty() {
943        quote!(spacetimedb::UniqueConstraintViolation)
944    } else {
945        quote!(::core::convert::Infallible)
946    };
947    let autoinc_err = if !sequenced_columns.is_empty() {
948        quote!(spacetimedb::AutoIncOverflow)
949    } else {
950        quote!(::core::convert::Infallible)
951    };
952
953    let field_types = fields.iter().map(|f| f.ty).collect::<Vec<_>>();
954
955    let tabletype_impl = quote! {
956        use spacetimedb::Serialize;
957        impl spacetimedb::Table for #tablehandle_ident {
958            type Row = #row_type;
959
960            type UniqueConstraintViolation = #unique_err;
961            type AutoIncOverflow = #autoinc_err;
962
963            #integrate_generated_columns
964        }
965        impl spacetimedb::table::TableInternal for #tablehandle_ident {
966            const TABLE_NAME: &'static str = #table_name;
967            // the default value if not specified is Private
968            #(const TABLE_ACCESS: spacetimedb::table::TableAccess = #table_access;)*
969            const UNIQUE_COLUMNS: &'static [u16] = &[#(#unique_col_ids),*];
970            const INDEXES: &'static [spacetimedb::table::IndexDesc<'static>] = &[#(#index_descs),*];
971            #(const PRIMARY_KEY: Option<u16> = Some(#primary_col_id);)*
972            const SEQUENCES: &'static [u16] = &[#(#sequence_col_ids),*];
973            #(const SCHEDULE: Option<spacetimedb::table::ScheduleDesc<'static>> = Some(#schedule);)*
974
975            #table_id_from_name_func
976            #default_fn
977        }
978    };
979
980    let register_describer_symbol = format!("__preinit__20_register_describer_{table_ident}");
981
982    let describe_table_func = quote! {
983        #[export_name = #register_describer_symbol]
984        extern "C" fn __register_describer() {
985            spacetimedb::rt::register_table::<#tablehandle_ident>()
986        }
987    };
988
989    // Output all macro data
990    let trait_def = quote_spanned! {table_ident.span()=>
991        #[allow(non_camel_case_types, dead_code)]
992        #vis trait #table_ident {
993            fn #table_ident(&self) -> &#tablehandle_ident;
994        }
995        impl #table_ident for spacetimedb::Local {
996            fn #table_ident(&self) -> &#tablehandle_ident {
997                &#tablehandle_ident {}
998            }
999        }
1000    };
1001
1002    let trait_def_view = quote_spanned! {table_ident.span()=>
1003        #[allow(non_camel_case_types, dead_code)]
1004        #vis trait #view_trait_ident {
1005            fn #table_ident(&self) -> &#viewhandle_ident;
1006        }
1007        impl #view_trait_ident for spacetimedb::LocalReadOnly {
1008            #[inline]
1009            fn #table_ident(&self) -> &#viewhandle_ident {
1010                &#viewhandle_ident {}
1011            }
1012        }
1013    };
1014
1015    let cols_struct_fields = fields.iter().map(|col| {
1016        let ident = col.ident.unwrap();
1017        let ty = &col.ty;
1018
1019        quote! {
1020            pub #ident: spacetimedb::query_builder::Col<#original_struct_ident, #ty>,
1021        }
1022    });
1023
1024    let ix_cols_struct_fields = indices.iter().filter_map(|index| {
1025        let ident = index.accessor_name.clone();
1026        let ty = index.kind.one_col()?.ty;
1027
1028        Some(quote! {
1029            pub #ident: spacetimedb::query_builder::IxCol<#original_struct_ident, #ty>,
1030        })
1031    });
1032
1033    let cols_init = fields.iter().map(|col| {
1034        let ident = col.ident.as_ref().unwrap();
1035
1036        quote! {
1037            #ident: spacetimedb::query_builder::Col::new(_table_name, stringify!(#ident)),
1038        }
1039    });
1040
1041    let ix_cols_init = indices.iter().map(|index| {
1042        if index.kind.one_col().is_none() {
1043            quote! {}
1044        } else {
1045            let ident = index.accessor_name;
1046            quote! {
1047                #ident: spacetimedb::query_builder::IxCol::new(_table_name, stringify!(#ident)),
1048            }
1049        }
1050    });
1051
1052    let query_builder_helper_structs = quote_spanned! {table_ident.span()=>
1053           #[allow(non_camel_case_types, dead_code)]
1054           pub struct #query_cols_struct{
1055               #(#cols_struct_fields)*
1056           }
1057
1058           impl spacetimedb::query_builder::HasCols for #original_struct_ident  {
1059               type Cols = #query_cols_struct;
1060                fn cols(_table_name: &'static str) -> Self::Cols {
1061                     #query_cols_struct {
1062                          #(#cols_init)*
1063                     }
1064                }
1065           }
1066
1067        #[allow(non_camel_case_types, dead_code)]
1068        pub struct #query_ix_cols_struct{
1069            #(#ix_cols_struct_fields)*
1070        }
1071        impl spacetimedb::query_builder::HasIxCols for #original_struct_ident {
1072            type IxCols = #query_ix_cols_struct;
1073            fn ix_cols(_table_name: &'static str) -> Self::IxCols {
1074                #query_ix_cols_struct {
1075                    #(#ix_cols_init)*
1076                }
1077            }
1078        }
1079
1080    };
1081
1082    let table_query_handle_def = quote! {
1083           #[allow(non_camel_case_types, dead_code)]
1084           #vis trait #query_trait_ident {
1085               fn #table_ident(&self) -> spacetimedb::query_builder::Table<#original_struct_ident> {
1086                   spacetimedb::query_builder::Table::new(stringify!(#table_ident))
1087               }
1088           }
1089           impl #query_trait_ident for spacetimedb::QueryBuilder {}
1090    };
1091
1092    let tablehandle_def = quote! {
1093        #[allow(non_camel_case_types)]
1094        #[non_exhaustive]
1095        #vis struct #tablehandle_ident {}
1096    };
1097
1098    let viewhandle_def = quote! {
1099        #[allow(non_camel_case_types)]
1100        #[non_exhaustive]
1101        #vis struct #viewhandle_ident {}
1102    };
1103
1104    let struct_name = original_struct_ident.to_string();
1105    let is_first_table = is_first_appearance(&struct_name);
1106
1107    let struct_level_query_impl = if is_first_table {
1108        quote! { #query_builder_helper_structs }
1109    } else {
1110        quote! {}
1111    };
1112
1113    let emission = quote! {
1114        const _: () = {
1115            #(let _ = <#field_types as spacetimedb::rt::TableColumn>::_ITEM;)*
1116            #schedule_typecheck
1117            #default_type_check
1118        };
1119
1120        #trait_def
1121        #trait_def_view
1122
1123        #tablehandle_def
1124        #viewhandle_def
1125        #table_query_handle_def
1126        #struct_level_query_impl
1127
1128        const _: () = {
1129            impl #tablehandle_ident {
1130                #(#index_accessors_rw)*
1131            }
1132
1133            impl #viewhandle_ident {
1134                #(#index_accessors_ro)*
1135            }
1136
1137            #tabletype_impl
1138
1139            #[allow(non_camel_case_types)]
1140            mod __indices {
1141                #[allow(unused)]
1142                use super::*;
1143                #(#index_marker_types)*
1144            }
1145
1146            #describe_table_func
1147        };
1148    };
1149
1150    if std::env::var("PROC_MACRO_DEBUG").is_ok() {
1151        {
1152            #![allow(clippy::disallowed_macros)]
1153            println!("{emission}");
1154        }
1155    }
1156
1157    Ok(emission)
1158}
