95/// Create or upsert a pipeline state record.
96/// Used by temporal_sync.py for filesystem reconciliation.
97#[reducer]
98pub fn upsert_pipeline(
99    ctx: &ReducerContext,
100    version: String,
101    status: String,
102    current_stage: String,
103    current_agent: String,
104    tags: String,
105    priority: String,
106) {
107    if let Some(existing) = pipeline_state::version().find(&version) {
108        pipeline_state::version().update(PipelineState {
109            status,
110            current_stage,
111            current_agent,
112            updated_at: ctx.timestamp,
113            tags,
114            priority,
115            ..existing
116        });
117    } else {
118        PipelineState::insert(PipelineState {
119            version,
120            status,
121            current_stage,
122            current_agent,
123            locked_by: None,
124            lock_acquired_at: None,
125            created_at: ctx.timestamp,
126            updated_at: ctx.timestamp,
127            tags,
128            priority,
129        });
130    }
131}

162/// Advance pipeline state to next stage + create handoff record.
163/// FLAG-2 fix: separated from transition logging.
164/// Called AFTER log_transition for atomic stage advancement.
165#[reducer]
166pub fn advance_pipeline(
167    ctx: &ReducerContext,
168    version: String,
169    completed_stage: String,
170    next_stage: String,
171    source_agent: String,
172    target_agent: String,
173    notes: String,
174) -> Result<(), String> {
175    let state = pipeline_state::version().find(&version)
176        .ok_or_else(|| format!("Pipeline {} not found", version))?;
177
178    // Update pipeline state
179    pipeline_state::version().update(PipelineState {
180        current_stage: next_stage.clone(),
181        current_agent: target_agent.clone(),
182        locked_by: None,
183        lock_acquired_at: None,
184        updated_at: ctx.timestamp,
185        ..state
186    });
187
188    // Create handoff record
189    Handoff::insert(Handoff {
190        id: 0,
191        version,
192        source_agent,
193        target_agent,
194        completed_stage,
195        next_stage,
196        notes,
197        dispatched_at: ctx.timestamp,
198        verified_at: None,
199        status: "dispatched".into(),
200        dispatch_payload_hash: String::new(),
201    });
202
203    Ok(())
204}

206/// Verify a handoff was acknowledged by the target agent.
207#[reducer]
208pub fn verify_handoff(
209    ctx: &ReducerContext,
210    handoff_id: u64,
211) -> Result<(), String> {
212    let h = handoff::id().find(&handoff_id)
213        .ok_or_else(|| format!("Handoff {} not found", handoff_id))?;
214
215    handoff::id().update(Handoff {
216        verified_at: Some(ctx.timestamp),
217        status: "verified".into(),
218        ..h
219    });
220
221    Ok(())
222}

233) {

241        agent_presence::agent().update(AgentPresence {
242            status,
243            current_pipeline: pipeline,
244            current_stage: stage,
245            last_heartbeat: ctx.timestamp,
246            session_id,
247            ..existing
248        });
249    } else {
250        AgentPresence::insert(AgentPresence {
251            agent,
252            status,
253            current_pipeline: pipeline,
254            current_stage: stage,
255            last_heartbeat: ctx.timestamp,
256            session_id,
257        });
258    }
259}

261/// Accumulate agent context using RFC 7396 JSON Merge Patch semantics.
262/// FLAG-4 fix: well-defined merge behavior.
263///
264/// Merge rules (JSON Merge Patch / RFC 7396):
265/// - Object keys: recursively merge nested objects
266/// - Arrays: concatenate (extension to RFC 7396 for our use case —
267///   we want to ACCUMULATE decisions/flags/questions, not replace them)
268/// - Null values: remove the key
269/// - Primitive values: new overwrites old
270#[reducer]
271pub fn update_agent_context(
272    ctx: &ReducerContext,
273    version: String,
274    agent: String,
275    context_delta: String,
276    session_id: String,
277    tokens_used: u64,
278) {
279    let key = format!("{}:{}", version, agent);
280
281    if let Some(existing) = agent_context::key().find(&key) {
282        let merged = merge_json_rfc7396_with_array_concat(
283            &existing.accumulated_context,
284            &context_delta,
285        );
286        agent_context::key().update(AgentContext {
287            accumulated_context: merged,
288            session_count: existing.session_count + 1,
289            total_tokens_used: existing.total_tokens_used + tokens_used,
290            last_session_id: session_id,
291            last_active_at: ctx.timestamp,
292            ..existing
293        });
294    } else {
295        AgentContext::insert(AgentContext {
296            key,
297            version,
298            agent,
299            accumulated_context: context_delta,
300            session_count: 1,
301            total_tokens_used: tokens_used,
302            last_session_id: session_id,
303            last_active_at: ctx.timestamp,
304            created_at: ctx.timestamp,
305        });
306    }
307}

339/// JSON Merge Patch (RFC 7396) with array concatenation extension.
340///
341/// - Objects: recursively merge
342/// - Arrays: concatenate (not replace) — extension for accumulating
343///   lists of decisions, flags, questions across sessions
344/// - Null in patch: delete the key from target
345/// - Primitives: patch value overwrites target value
346fn merge_json_rfc7396_with_array_concat(base: &str, patch: &str) -> String {
347    let base_val: serde_json::Value = serde_json::from_str(base)
348        .unwrap_or(serde_json::Value::Object(serde_json::Map::new()));
349    let patch_val: serde_json::Value = serde_json::from_str(patch)
350        .unwrap_or(serde_json::Value::Object(serde_json::Map::new()));
351
352    let merged = merge_values(base_val, patch_val);
353    serde_json::to_string(&merged).unwrap_or_else(|_| "{}".to_string())
354}
355
356fn merge_values(base: serde_json::Value, patch: serde_json::Value) -> serde_json::Value {
357    use serde_json::Value;
358
359    match (base, patch) {
360        // Both objects: recursively merge keys
361        (Value::Object(mut base_map), Value::Object(patch_map)) => {
362            for (key, patch_val) in patch_map {
363                if patch_val.is_null() {
364                    // RFC 7396: null means delete the key
365                    base_map.remove(&key);
366                } else if let Some(base_val) = base_map.remove(&key) {
367                    base_map.insert(key, merge_values(base_val, patch_val));
368                } else {
369                    base_map.insert(key, patch_val);
370                }
371            }
372            Value::Object(base_map)
373        }
374        // Both arrays: concatenate (extension to RFC 7396)
375        (Value::Array(mut base_arr), Value::Array(patch_arr)) => {
376            base_arr.extend(patch_arr);
377            Value::Array(base_arr)
378        }
379        // All other cases: patch overwrites base
380        (_, patch) => patch,
381    }
